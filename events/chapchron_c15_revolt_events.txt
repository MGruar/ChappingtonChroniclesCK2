# Tributary Rebellions
character_event = {
	id = ChapChron.100
	title = EVTNAME_CHAPCHRON_100
	desc = EVTDESC_CHAPCHRON_100
	
	picture = GFX_evt_peasants
	border = GFX_event_normal_frame_war
	
	major = yes
	major_trigger = {
		ai = no
	}
	
	only_playable = yes
	
	trigger = {
		year = 1410
		NOT = {
			has_character_flag = chapchron_c15 # legacy compatability
			has_character_flag = chapchron_c15_tributary_rebellion
			year = 1440
		}
		is_tributary = yes
		suzerain = {
			has_landed_title = e_britannia
		}
		is_incapable = no
	}
	
	immediate = {
		set_character_flag = chapchron_c15_tributary_rebellion
		ROOT = {
			suzerain = {
				reverse_war = {
					target = PREV
					casus_belli = free_tributary_cb
				}
				opinion = {
					who = PREV
					modifier = opinion_evil_tyrant
				}
			}
		}
		# Spawn more troops
		spawn_unit = {
			province = THIS
			home = THIS
			owner = THIS
			#leader = THIS
			troops = {
				archers = { 5000 5000 }
				light_infantry = { 15000 15000 }
				heavy_infantry = { 3000 3000 }
				light_cavalry = { 2000 2000 }
			}
			attrition = 0.25
		}
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = EVTOPTA_CHAPCHRON_100
	}
}

# Nationalist Uprising 1
province_event = {
	id = ChapChron.101
	title = EVTNAME_CHAPCHRON_101
	desc = EVTDESC_CHAPCHRON_101
	
	picture = GFX_evt_peasants
	border = GFX_event_normal_frame_war
	
	major = yes
	major_trigger = {
		ai = no
	}
	
	trigger = {
		year = 1415
		NOT = {
			has_global_flag = chapchron_c15_nationalist_uprising
			year = 1440
			culture_group = celtic
		}
		owner = {
			top_liege = {
				has_landed_title = e_britannia
			}
		}
	}
	
	immediate = {
		set_global_flag = chapchron_c15_nationalist_uprising
		
		create_character = {
			random_traits = yes
			dynasty = random
			religion = ROOT
			culture = ROOT
			female = no
			age = 20
			health = 7
			fertility = 0.7
			attributes = {
				martial = 8
				diplomacy = 9
			}
			trait = just
			trait = brave
			trait = gregarious
			trait = ambitious
			trait = brilliant_strategist
			trait = inspiring_leader
		}
		
		new_character = {
			set_character_flag = nationalist_rebel
			
			create_title = {
				tier = DUKE
				landless = yes
				temporary = yes
				rebel = yes
				culture = ROOT
				name = "NATIONALIST_REVOLT"
				holder = THIS
			}
			
			wealth = 1000
			
			# DoW on the province top liege
			ROOT = {
				owner = {
					top_liege = {
						reverse_war = {
							target = PREVPREVPREV
							casus_belli = chapchron_nationalist_revolt
							thirdparty_title = ROOT # The county...
							tier = KING # ... but really the de jure kingdom
						}
						reverse_opinion = {
							who = PREVPREVPREV
							modifier = opinion_evil_tyrant
						}
						spawn_unit = {
							province = ROOT
							home = ROOT
							owner = PREVPREVPREV
							#leader = PREVPREVPREV
							match_character = THIS
							match_mult = 0.80
							match_min = 1000
							match_max = 80000
							attrition = 0.25
						}
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 48
	}
	
	option = {
		name = EVTOPTA_CHAPCHRON_101
	}
}

			home = THIS

			troops = {
				archers = { 5000 5000 }
				light_infantry = { 15000 15000 }
				heavy_infantry = { 3000 3000 }
				light_cavalry = { 2000 2000 }
			}
			attrition = 0.25
		}
	}

		name = EVTOPTA_CHAPCHRON_102
	}
}
