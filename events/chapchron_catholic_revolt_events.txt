# The Iberian Revolt Part 1
province_event = {
	id = ChapChron.0
	title = EVTNAME_CHAPCHRON_0
	desc = EVTDESC_CHAPCHRON_0
	
	picture = GFX_evt_heretic
	border = GFX_event_normal_frame_war
	
	trigger = {
		year = 1315
		province_id = 157 # Santiago
		NOT = {
			has_global_flag = chapchron_c14_iberia_1
		}
	}
	
	immediate = {
		set_global_flag = chapchron_c14_iberia_1
		# Find a Catholic Chappington, create a revolt here
		ROOT = {
			owner = {
				top_liege = {
					random_dynasty_member = {
						limit = {
							religion = catholic
							culture = scottish
						}
						
						set_character_flag = religious_revolter
						
						create_title = {
							tier = DUKE
							landless = yes
							temporary = yes
							rebel = yes
							culture = ROOT
							name = "HERETIC_REVOLT"
							holder = THIS
						}
						
						wealth = 1000
						
						spawn_unit = {
							province = ROOT
							home = ROOT
							owner = THIS
							#leader = THIS
							match_character = PREV
							match_mult = 0.25
							match_min = 1000
							match_max = 50000
							attrition = 0.25
						}
						
						# DoW on the province top liege
						war = {
							target = PREV
							casus_belli = religious_revolt
							thirdparty_title = ROOT # The county...
							tier = DUKE # ... but really the de jure duchy
						}
						opinion = {
							who = PREV
							modifier = opinion_evil_tyrant
						}
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 720
	}
	
	option = {
		name = EVTOPTA_CHAPCHRON_0
	}
}

# The Highlander Revolt
province_event = {
	id = ChapChron.4
	title = EVTNAME_CHAPCHRON_4
	desc = EVTDESC_CHAPCHRON_4
	
	picture = GFX_evt_heretic
	border = GFX_event_normal_frame_war
	
	trigger = {
		year = 1315
		region = world_europe_west_brittania
		religion = catholic
		culture = scottish
		NOT = {
			has_global_flag = chapchron_c14_highland
		}
		has_global_flag = chapchron_c14_angsax_1
	}
	
	immediate = {
		set_global_flag = chapchron_c14_highland
		# Create a new character for the revolt.
		create_character = {
			random_traits = yes
			dynasty = none
			religion = ROOT
			culture = ROOT
			female = no
			age = 32
			attributes = {
				martial = 7
				learning = 7
			}
			trait = zealous
			trait = scholar
			trait = skilled_tactician
		}
		
		new_character = {
			set_character_flag = religious_revolter
			
			create_title = {
				tier = DUKE
				landless = yes
				temporary = yes
				rebel = yes
				culture = ROOT
				name = "HERETIC_REVOLT"
				holder = THIS
			}
			
			wealth = 1000
			
			# DoW on the province top liege
			ROOT = {
				owner = {
					top_liege = {
						reverse_war = {
							target = PREVPREVPREV
							casus_belli = religious_revolt

							tier = DUKE # ... but really the de jure duchy
						}
						reverse_opinion = {
							who = PREVPREVPREV
							modifier = opinion_evil_tyrant
						}
						spawn_unit = {
							province = ROOT
							home = ROOT

							match_mult = 0.35
							match_min = 1000
							match_max = 50000
							attrition = 0.25
						}
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 6
	}
	
	option = {
		name = EVTOPTA_CHAPCHRON_4
	}
}
